\begin{block}{Topologie}
Soit $n = O(\max{(a,b)})$, la profondeur de développement 
en fractions continues de $a/b$, la pente du motif. 
\scriptsize
 \begin{itemize}
  \item $n+2$ sommets
  \item $2n+1$ arêtes ($n+2$ au bord et $n-1$ internes)
  \item $n$ faces
 \end{itemize}
\end{block}

\begin{frame}
  	\frametitle{Retour sur l'algorithme d'Euclide}

%equations
Soit $\theta(z_n) = p_n/q_n = [u_0, u_1, \ldots, u_n]$
($q_n > p_n > 0$ sont des entiers premiers entre eux). 

\begin{equation}
\label{eq:rec-rem}
\begin{array}{l}
  u_0 = 0, \quad r_{-1} = q_n, \: r_0 = p_n, \\
  \forall 1 \leq k \leq n, \:
  u_k = \bigg\lfloor\frac{r_{k-2}}{r_{k-1}}\bigg\rfloor, \:
  r_k = r_{k-2} - u_k r_{k-1}. \\
\end{array}
\end{equation}

Les restes sont positifs et décroissants.  
\begin{equation}
\label{eq:rem-order}
r_{-1} = q_n > p_n = r_{0} > \ldots > r_{n-1} = 1 > r_n = 0. 
\end{equation}


Les convergents $z_k$ sont définis tels que $\forall 0 \leq k \leq n$, 
$p_k/q_k$ $=$ $[u_0, u_1, \ldots, u_k]$. 

\begin{equation}
\label{eq:rec-conv}
\begin{array}{l}
  z_0 = 1, \: z_{-1} = i, \\
  \forall 1 \leq k \leq n, \:
  z_k = z_{k-2} + u_k z_{k-1}. \\
\end{array}
 \quad
%
\end{equation}

\end{frame}

\begin{equation}
\label{eq:det-k1}
\forall 0 \leq k \leq n \quad 
q_kp_{k-1} - p_{k-1}q_k = (-1)^k.
\end{equation}
%
\begin{equation}
\label{eq:det-k2}
\forall 1 \leq k \leq n \quad 
q_kp_{k-2} - p_{k-2}q_k = -u_k(-1)^k.
\end{equation}
%
\begin{equation}
\label{eq:det-rem}
\forall -1 \leq k \leq n \quad 
q_np_k - p_kq_n = r_k(-1)^k.
\end{equation}

\begin{frame}
  	\frametitle{Définition de la triangulation}

%ordre
\scriptsize
$(H_0, O, Z_n)$, $(H_1, H_0, O)$, $\ldots$, 
$(H_k, H_{k-1}, H_{k-2})$, $\ldots$, 
$(H_{n-1}, H_{n-2}, H_{n-3})$. 
\normalsize

\begin{figure}
\centering
\includegraphics[width=0.4\linewidth,page=2]{fig/FDT-schema-bis.pdf}\hspace{0.05\linewidth}
\end{figure}

\begin{block}{Idée de la preuve}
 \begin{itemize}
  \item I. c'est une triangulation
  \item II. le cercle circonscrit de chaque triangle contient tous les points
  \begin{enumerate}
   \item \alert{deux triangles consécutifs partagent une arête; le cercle circonscrit 
du second contient le sommet opposé du premier}
   \item conclure par récurrence 
  \end{enumerate}
 \end{itemize}
\end{block}

\end{frame}

\begin{frame}
  	\frametitle{Exemple ($k < n$, $k$ impair)}

\begin{block}{Trois équivalences}
 \begin{enumerate}
  \item Le cercle passant par $H_k$, $H_{k-1}$, $H_{k-2}$ contient $H_{k-3}$.
  \item $(\vect{H_{k-2}H_{k}},\vect{H_{k-2}H_{k-1}}) < (\vect{H_{k-3}H_k},\vect{H_{k-3}H_{k-1}})$
  \item \scriptsize{
$x = (z_n - z_{k-1} - z_{k-2})\conj{z_{k-1}}$,
$y = \conj{(z_n - z_{k} - z_{k-3})}z_{k-2}$ }
\normalsize{ et $\theta(x) < \theta(y)$.} 
 \end{enumerate}
\end{block}

\centering
\includegraphics[width=0.3\linewidth]{angles.pdf}

\begin{block}{3 est vrai car: }
 \begin{itemize}
  \item (parties réelles) $\Re(x) > \Re(y) > 0$, par (\ref{eq:rec-conv}). 
  \item (parties imaginaires) 
Par (\ref{eq:rec-conv}), (\ref{eq:rem-conv}), 
(\ref{eq:det-k1}), (\ref{eq:det-k2}):
  \begin{itemize}
   \item $\Im(x) = (r_{k-1} - 1)$.
   \item $\Im(y) = (r_{k-1} - 1)u_k + (r_k - 1)$. 
   \item D'où $\Im(y) \geq \Im(x) > 0$, par (\ref{eq:rem-order}). 
  \end{itemize}
 \end{itemize}
\end{block}

\end{frame}
